type node_t
    integer :: next = NULL_NODE
    integer :: parent = NULL_NODE
    integer :: left = NULL_NODE
    integer :: right = NULL_NODE
    integer :: height = -1
    integer :: atom = 0
    type(aabb_t) :: aabb
    contains
        procedure :: init => node_init
        procedure :: isleaf => node_isleaf
        procedure :: asstr => node_asstr
end type node_t
